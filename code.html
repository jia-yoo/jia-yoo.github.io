interface Developer {
    void writeCode();
    void debugCode();
}

interface JuniorDeveloper extends Developer {
    void learn();
}

interface Dedicated {
    void showDedication();
}

interface QuickLearner {
    void quickLearner();
}

abstract class BaseDeveloper implements Developer {
    String name;
    String email;

    BaseDeveloper(String name, String email) {
        this.name = name;
        this.email = email;
    }

    @Override
    public void writeCode() {
        System.out.println(name + " is writing code.");
    }

    @Override
    public void debugCode() {
        System.out.println(name + " is debugging code.");
    }

    abstract void introduce();
}

class JuniorJavaDeveloper extends BaseDeveloper implements JuniorDeveloper, Dedicated, QuickLearner {
    String favoriteLanguage;
    boolean fluentInEnglish;

    JuniorJavaDeveloper(String name, String email, String favoriteLanguage, boolean fluentInEnglish) {
        super(name, email);
        this.favoriteLanguage = favoriteLanguage;
        this.fluentInEnglish = fluentInEnglish;
    }

    @Override
    public void learn() {
        System.out.println(name + " is eager to learn new technologies and improve skills.");
    }

    public void showDedication() {
        System.out.println(name + " is highly dedicated and hardworking.");
    }

    public void quickLearner() {
        System.out.println(name + " has the ability to learn quickly and adapt to new environments.");
    }

    public void showLanguageSkills() {
        if (fluentInEnglish) {
            System.out.println(name + " is fluent in English.");
        }
    }

    @Override
    void introduce() {
        
        showDedication();
        learn();
        quickLearner();
        showLanguageSkills();
    }
}

public class Main {
    public static void main(String[] args) {
        JuniorJavaDeveloper developer = new JuniorJavaDeveloper("John Doe", "john.doe@example.com", "Java", true);
        developer.introduce();
        developer.writeCode();
        developer.debugCode();
    }
