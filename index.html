    static class JuniorDeveloper extends BaseDeveloper implements QuickLearner {
        String favoriteLanguage;
        boolean fluentInEnglish;
 
        JuniorDeveloper(String name, String favoriteLanguage, boolean fluentInEnglish) {
            super(name);
            this.favoriteLanguage = favoriteLanguage;
            this.fluentInEnglish = fluentInEnglish;
        }
 
        @Override
        void introduce() {
          System.out.println("안녕하세요 :) 주니어 백엔드 개발자 " + name + "입니다.");
          System.out.println("코딩할 때는 " + favoriteLanguage + ", 대화할 때는 영어, 둘 다 제가 좋아하는 언어입니다!");
        }
 
        @Override
        public void writeAndDebugCode() {
            System.out.println("문제를 파악해 해결책을" + favoriteLanguage + "코드로 구현할 수 있으며, 오류를 분석해 문제를 수정할 수 있습니다.");
        }
    }
 
    public static void main(String[] args) {
        JuniorDeveloper developer = new JuniorDeveloper("유지아", "Java", true);
        developer.introduce();
        developer.writeAndDebugCode();
    }

About Me

유지아

성실함을 바탕으로 꾸준히 성장하는 개발자

6년 간의 호주 생활을 통해, 다양한 문화권의 사람들과 협업하며 비즈니스 영어 능력을 포함한 효율적인 의사소통 능력을 갖추게 되었습니다.
맡은 업무를 책임감 있게 성실히 수행하는 모습을 인정받아, 로컬 중심의 회사에서 첫 외국인 Supervisor로 승진하며 팀을 이끌 수 있는 기회도 얻을 수 있었습니다.

세 번의 팀 프로젝트를 통해 협업 능력과 기술적 문제 해결 역량을 크게 발전시켰습니다. 첫 번째 프로젝트에서는 일정 지연에도 불구하고 더 많은 부분을 담당해 완수했고,
두 번째, 세 번째 프로젝트에서는 기술 지원 역할을 맡아 팀원들의 문제를 함께 해결하고, 나아가 새로운 기술을 빠르게 학습하고 구현하여 일정보다 앞서 업무를 완료한 경험이 있습니다.
이러한 경험을 통해 배운 성실함과 능동적인 태도로 회사의 발전에 기여하고, 꾸준히 성장하는 개발자가 되겠습니다.
